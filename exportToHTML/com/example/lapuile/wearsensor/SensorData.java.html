88       @Override 
89       public void onSensorChanged(SensorEvent event) { 
90    
91           int sensorType = event.sensor.getType(); 
92    
93           switch (sensorType){ 
94               case Sensor.TYPE_ACCELEROMETER: 
95                   // In this example, alpha is calculated as t / (t + dT), 
96                   // where t is the low-pass filter's time-constant and 
97                   // dT is the event delivery rate. 
98    
99                   final float alpha = (float)0.8; 
100                  float[] gravity=new float[3]; 
101                  float[] linear_acceleration = new float[3]; 
102   
103                  // Isolate the force of gravity with the low-pass filter. 
104                  gravity[0] = alpha * gravity[0] + (1 - alpha) * event.values[0]; 
105                  gravity[1] = alpha * gravity[1] + (1 - alpha) * event.values[1]; 
106                  gravity[2] = alpha * gravity[2] + (1 - alpha) * event.values[2]; 
107   
108                  // Remove the gravity contribution with the high-pass filter. 
109                  linear_acceleration[0] = event.values[0] - gravity[0]; 
110                  linear_acceleration[1] = event.values[1] - gravity[1]; 
111                  linear_acceleration[2] = event.values[2] - gravity[2]; 
112   
113                  TextView acceleration_text = findViewById(R.id.sensor_list); 
114                  StringBuilder acc_text = new StringBuilder(); 
115                  acc_text.append(getResources().getString(R.string.accelerometerx_text, linear_acceleration[0])).append( 
116                          System.getProperty("line.separator")) 
117                          .append(getResources().getString(R.string.accelerometery_text, linear_acceleration[1])).append( 
118                      System.getProperty("line.separator")).append( 
119                 getResources().getString(R.string.accelerometerz_text, linear_acceleration[2])); 
120   
121                  acceleration_text.setText(acc_text); 
122          }
